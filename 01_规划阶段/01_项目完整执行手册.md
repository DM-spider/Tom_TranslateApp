### 2.3 翻译引擎实现

**`app/services/engines/base.py`** — 引擎基类：

```python
from abc import ABC, abstractmethod
from pydantic import BaseModel
from enum import Enum


class EngineType(str, Enum):
    DEEPSEEK = "deepseek"
    OPENAI = "openai"
    BAIDU = "baidu"


class TranslateRequest(BaseModel):
    texts: list[str]
    source_lang: str = "auto"
    target_lang: str = "zh-CN"
    engine: EngineType = EngineType.DEEPSEEK


class TranslateResult(BaseModel):
    translated_texts: list[str]
    source_lang: str
    engine_used: str
    tokens_used: int | None = None


class BaseEngine(ABC):
    engine_name: str

    @abstractmethod
    async def translate(
        self,
        texts: list[str],
        source_lang: str,
        target_lang: str,
    ) -> TranslateResult:
        ...
```

**`app/services/engines/deepseek.py`** — DeepSeek 引擎（核心，最先实现）：

```python
from openai import AsyncOpenAI
from app.config import get_settings
from .base import BaseEngine, TranslateResult

SEP = "\n[SEP]\n"

SYSTEM_PROMPT = """You are a professional translator.
Translate the user's text from {source_lang} to {target_lang}.
Rules:
- Produce natural, fluent, idiomatic translations
- Maintain the original formatting
- Keep proper nouns, brand names, and technical terms unchanged when appropriate
- If the input contains segments separated by [SEP], translate each segment independently and keep [SEP] as the separator in your output
- Output ONLY the translation, no explanations or notes"""


class DeepSeekEngine(BaseEngine):
    engine_name = "deepseek"

    def __init__(self):
        settings = get_settings()
        self.client = AsyncOpenAI(
            api_key=settings.deepseek_api_key,
            base_url=settings.deepseek_base_url,
        )
        self.model = settings.deepseek_model

    async def translate(
        self, texts: list[str], source_lang: str, target_lang: str
    ) -> TranslateResult:
        combined = SEP.join(texts)
        lang_display = source_lang if source_lang != "auto" else "the detected language"

        response = await self.client.chat.completions.create(
            model=self.model,
            messages=[
                {
                    "role": "system",
                    "content": SYSTEM_PROMPT.format(
                        source_lang=lang_display, target_lang=target_lang
                    ),
                },
                {"role": "user", "content": combined},
            ],
            temperature=0.3,
        )

        result_text = response.choices[0].message.content.strip()
        translated_texts = [t.strip() for t in result_text.split("[SEP]")]

        if len(translated_texts) != len(texts):
            translated_texts = (translated_texts + [""] * len(texts))[: len(texts)]

        return TranslateResult(
            translated_texts=translated_texts,
            source_lang=source_lang,
            engine_used=self.engine_name,
            tokens_used=response.usage.total_tokens if response.usage else None,
        )
```

**`app/services/engines/baidu.py`** — 百度翻译（兜底引擎）：

```python
import hashlib
import random
import httpx
from .base import BaseEngine, TranslateResult
from app.config import get_settings

BAIDU_API_URL = "https://fanyi-api.baidu.com/api/trans/vip/translate"

LANG_MAP = {
    "zh-CN": "zh", "en": "en", "ja": "jp", "ko": "kor",
    "fr": "fra", "de": "de", "es": "spa", "ru": "ru",
    "auto": "auto",
}


class BaiduEngine(BaseEngine):
    engine_name = "baidu"

    def __init__(self):
        settings = get_settings()
        self.app_id = settings.baidu_app_id
        self.secret_key = settings.baidu_secret_key

    def _make_sign(self, query: str, salt: str) -> str:
        raw = self.app_id + query + salt + self.secret_key
        return hashlib.md5(raw.encode()).hexdigest()

    async def translate(
        self, texts: list[str], source_lang: str, target_lang: str
    ) -> TranslateResult:
        combined = "\n".join(texts)
        salt = str(random.randint(32768, 65536))

        params = {
            "q": combined,
            "from": LANG_MAP.get(source_lang, source_lang),
            "to": LANG_MAP.get(target_lang, target_lang),
            "appid": self.app_id,
            "salt": salt,
            "sign": self._make_sign(combined, salt),
        }

        async with httpx.AsyncClient() as client:
            resp = await client.get(BAIDU_API_URL, params=params)
            data = resp.json()

        if "trans_result" not in data:
            raise Exception(f"Baidu API error: {data}")

        translated_texts = [item["dst"] for item in data["trans_result"]]
        detected_lang = data.get("from", source_lang)

        return TranslateResult(
            translated_texts=translated_texts,
            source_lang=detected_lang,
            engine_used=self.engine_name,
            tokens_used=None,
        )
```

### 2.4 翻译服务（调度层 + 缓存）

**`app/services/cache.py`**：

```python
import json
import hashlib
import redis.asyncio as redis
from app.config import get_settings


class TranslateCache:
    def __init__(self):
        self.redis = redis.from_url(get_settings().redis_url)
        self.ttl = 86400 * 7  # 缓存 7 天

    def _make_key(self, text: str, source: str, target: str, engine: str) -> str:
        raw = f"{engine}:{source}:{target}:{text}"
        return f"trans:{hashlib.sha256(raw.encode()).hexdigest()}"

    async def get(self, text: str, source: str, target: str, engine: str) -> str | None:
        key = self._make_key(text, source, target, engine)
        result = await self.redis.get(key)
        return result.decode() if result else None

    async def set(self, text: str, source: str, target: str, engine: str, translated: str):
        key = self._make_key(text, source, target, engine)
        await self.redis.setex(key, self.ttl, translated)

    async def get_batch(
        self, texts: list[str], source: str, target: str, engine: str
    ) -> list[str | None]:
        keys = [self._make_key(t, source, target, engine) for t in texts]
        results = await self.redis.mget(keys)
        return [r.decode() if r else None for r in results]
```

**`app/services/translator.py`** — 核心调度逻辑：

```python
from .engines.base import BaseEngine, EngineType, TranslateRequest, TranslateResult
from .engines.deepseek import DeepSeekEngine
from .engines.baidu import BaiduEngine
from .cache import TranslateCache


class TranslatorService:
    def __init__(self):
        self.engines: dict[str, BaseEngine] = {
            EngineType.DEEPSEEK: DeepSeekEngine(),
            EngineType.BAIDU: BaiduEngine(),
        }
        self.cache = TranslateCache()

    async def translate(self, req: TranslateRequest) -> TranslateResult:
        engine = self.engines[req.engine]

        # 1. 批量查缓存
        cached = await self.cache.get_batch(
            req.texts, req.source_lang, req.target_lang, req.engine
        )

        # 2. 找出未命中缓存的文本
        uncached_indices = [i for i, c in enumerate(cached) if c is None]
        uncached_texts = [req.texts[i] for i in uncached_indices]

        # 3. 翻译未命中的部分
        if uncached_texts:
            result = await engine.translate(
                uncached_texts, req.source_lang, req.target_lang
            )
            # 4. 写入缓存
            for idx, translated in zip(uncached_indices, result.translated_texts):
                await self.cache.set(
                    req.texts[idx], req.source_lang, req.target_lang,
                    req.engine, translated
                )
                cached[idx] = translated
        else:
            result = TranslateResult(
                translated_texts=[],
                source_lang=req.source_lang,
                engine_used=req.engine,
                tokens_used=0,
            )

        return TranslateResult(
            translated_texts=cached,
            source_lang=result.source_lang,
            engine_used=req.engine,
            tokens_used=result.tokens_used,
        )
```

const API_BASE = process.env.NEXT_PUBLIC_API_URL || "http://localhost:8000";

export interface TranslateParams {
  texts: string[];
  sourceLang?: string;
  targetLang?: string;
  engine?: string;
}

export async function translateText(params: TranslateParams) {
  const res = await fetch(`${API_BASE}/api/v1/translate`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      texts: params.texts,
      source_lang: params.sourceLang || "auto",
      target_lang: params.targetLang || "zh-CN",
      engine: params.engine || "deepseek",
    }),
  });
  if (!res.ok) throw new Error("Translation failed");
  return res.json();
}

| news.ycombinator.com | 简单文本结构、嵌套评论 |
| twitter.com / x.com | 动态加载、虚拟列表 |
| medium.com | 富文本、图片与文字混排 |
| arxiv.org | 学术论文、数学公式需要保留 |
| amazon.com | 商品页面、多种布局混合 |

### 5.2 常见问题与解决方案

| 问题 | 解决方案 |
|------|---------|
| 动态加载内容（SPA）翻译不到 | 用 `MutationObserver` 监听 DOM 变化，新增节点自动翻译 |
| 译文导致排版变形 | 译文元素用 `display:block` 独立成行，不影响原文布局 |
| 翻译速度慢 | 实现懒翻译：先翻译视口内容，`IntersectionObserver` 监听滚动 |
| 大模型返回的分段数量不匹配 | 校验逻辑 + 回退：不匹配时逐条翻译 |
| 扩展与页面 CSS 冲突 | 使用 Shadow DOM 或极高特异性选择器隔离样式 |
